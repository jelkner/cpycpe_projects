Display a Number in Binary with Neopixels

By: Jeffrey Elkner

Introduction Bits and Binary Basics Python Bit Manipulation Final Product 
import time
from adafruit_circuitplayground.express import cpx

pixels = 10 * [False]

number = 17
mask = 1

for pixnum in range(9, -1, -1):
    pixels[pixnum] = bool(number & mask)
    mask <<= 1

cpx.pixels.brightness = 0.3
while True:
    for pos, is_one in enumerate(pixels):
        cpx.pixels[pos] = (0, 30, 0) if is_one else (30, 0, 0)
